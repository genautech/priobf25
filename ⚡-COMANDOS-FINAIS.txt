🚀 EXECUTE LINHA POR LINHA

# 1. Navegue até a pasta (AJUSTE O CAMINHO SE NECESSÁRIO)
cd Downloads/prio/priobf25

# 2. Verifique se está na pasta certa
ls

# 3. Adicione o remote do GitHub
git remote add origin https://github.com/genautech/priobf25.git

# 4. Configure seu nome (SUBSTITUA "Seu Nome")
git config user.name "Seu Nome"

# 5. Configure seu email (SUBSTITUA com seu email do GitHub)
git config user.email "seu-email@example.com"

# 6. Adicione TODOS os arquivos
git add -A

# 7. Faça o commit
git commit -m "deploy: v6.1.2.3 - sistema completo funcional, bugs corrigidos"

# 8. Baixe mudanças do GitHub (se houver)
git pull origin main --rebase

# 9. Envie para o GitHub
git push origin main

# ✅ PRONTO! Aguarde 2 minutos para deploy automático


═══════════════════════════════════════════════════════════════
🆘 SE DER ERRO "remote origin already exists"
═══════════════════════════════════════════════════════════════

# Remova o remote antigo
git remote remove origin

# Adicione novamente
git remote add origin https://github.com/genautech/priobf25.git

# Continue do passo 6 acima


═══════════════════════════════════════════════════════════════
🆘 SE DER ERRO "failed to push"
═══════════════════════════════════════════════════════════════

# Execute pull primeiro
git pull origin main --rebase

# Tente push novamente
git push origin main
