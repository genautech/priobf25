🚀 OPÇÃO 1: SE GIT JÁ ESTÁ CONFIGURADO
═══════════════════════════════════════════════════════════════

# Abra o terminal e execute linha por linha:

cd ~/downloads/prio/priobf25

git add -A

git pull origin main --rebase

git push origin main

# ✅ PRONTO! Aguarde 2 minutos para deploy automático


═══════════════════════════════════════════════════════════════
🔧 OPÇÃO 2: PRIMEIRA VEZ (CONFIGURAR TUDO)
═══════════════════════════════════════════════════════════════

# Execute linha por linha:

cd ~/downloads/prio/priobf25

git init

git remote add origin https://github.com/genautech/priobf25.git

git config user.name "Seu Nome"

git config user.email "seu-email@example.com"

git add -A

git commit -m "fix: v6.1.2.3 - sistema completo com bugs corrigidos"

git pull origin main --rebase

git push origin main

# ⚠️ GitHub pode pedir login:
# - Se abrir navegador: faça login
# - Se pedir senha: use TOKEN (não senha)
#   → Crie token em: https://github.com/settings/tokens


═══════════════════════════════════════════════════════════════
🐛 SE DER ERRO
═══════════════════════════════════════════════════════════════

# Erro: "failed to push"
git pull origin main --rebase
git push origin main

# Erro: "Permission denied"
git remote set-url origin https://github.com/genautech/priobf25.git
git push origin main

# Erro: "Conflitos ao fazer pull"
git checkout --ours .
git add -A
git rebase --continue
git push origin main
