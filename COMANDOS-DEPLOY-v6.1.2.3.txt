# → NÃO inicialize com README, .gitignore ou license

# PASSO 2: Configurar Git localmente
cd ~/downloads/prio/priobf25

# Inicializar Git
git init

# Configurar usuário (substitua com seus dados)
git config user.name "Seu Nome"
git config user.email "seu-email@example.com"

# Adicionar todos os arquivos
git add -A

# Primeiro commit
git commit -m "initial commit: versão v6.1.2.3 - sistema completo com bugs corrigidos"

# Criar branch main
git branch -M main

# Conectar ao GitHub (SUBSTITUA SEU-USUARIO pelo seu username)
git remote add origin https://github.com/SEU-USUARIO/priobf25.git

# Push inicial
git push -u origin main

# ✅ Arquivos enviados para GitHub!

# TROUBLESHOOTING
# ═══════════════════════════════════════════════════════════════

# Problema: "fatal: not a git repository"
# Solução:
cd ~/downloads/prio/priobf25
git init

# Problema: "fatal: remote origin already exists"
# Solução:
git remote remove origin
git remote add origin https://github.com/SEU-USUARIO/priobf25.git

# Problema: "failed to push some refs"
# Solução (pull primeiro):
git pull origin main --rebase
git push origin main

# Problema: "Permission denied (publickey)"
# Solução: Use HTTPS ao invés de SSH
git remote set-url origin https://github.com/SEU-USUARIO/priobf25.git
